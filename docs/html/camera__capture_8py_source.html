75  def contours(self, hsv, colorLower, colorUpper):
76  colorMask = cv2.inRange(hsv, colorLower, colorUpper)
77  colorMask = cv2.erode(colorMask, None, iterations=2)
78  colorMask = cv2.dilate(colorMask, None, iterations=2)
79  colorCnts = cv2.findContours(colorMask.copy(), cv2.RETR_EXTERNAL,
80  cv2.CHAIN_APPROX_SIMPLE)
81  colorCnts = imutils.grab_contours(colorCnts)
82  return colorCnts
